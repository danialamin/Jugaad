' GamePanel.checkZombieTouchEncounters detects player-zombie contact
' (collision-style, not menu-based). startZombieCombat() configures the
' enemy via ZombieEnemy(displayName, zoneId, hp), removes the zombie
' furniture, plays fight music, and shows a pre-fight dialogue.
'
' Player chooses Debug/Terminate from the combat overlay (UI). Both
' paths now include a mini-game (quiz for Debug, dodge mini-game for
' Terminate); the boolean "passedMinigame" is forwarded to the strategy
' through CombatController.executeCombatMove(move, passed, isBoss).

participant ":GamePanel" as gamePanel
participant ":ObjectManager" as objM
participant ":Furniture" as zFurn
participant ":ZombieEnemy" as zEnemy
participant ":CombatController" as combatController
participant ":SoundManager" as soundManager
participant ":UI" as gameUI
participant ":CombatStrategyFactory" as combatStrategyFactory
participant ":DebugStrategy" as debugStrategy
participant ":TerminateStrategy" as terminateStrategy
participant ":CombatChallenge" as combatChallenge
participant ":CombatResult" as combatResult
participant ":Player" as playerEntity
participant ":PlayerStats" as playerStats
participant ":KarmaTracker" as karmaTracker

[-> gamePanel: update()
activate gamePanel
gamePanel -> gamePanel: checkZombieTouchEncounters()
gamePanel -> objM: getZombieTouchingPlayer(...)
objM --> gamePanel: zFurn (or null)

opt zFurn != null
    gamePanel -> gamePanel: startZombieCombat(zFurn)
    gamePanel -> zEnemy **: new ZombieEnemy(displayName, zoneId, hp)
    gamePanel -> combatController: initiateCombat(zEnemy)
    gamePanel -> objM: furnitureList.remove(zFurn)
    gamePanel -> soundManager: playFightMusic(zombieName)
    gamePanel -> gameUI: startPreFightDialogue(name, hp)
    gamePanel -> gamePanel: phaseTwoState = IN_COMBAT\ngameState = dialogueState
end
deactivate gamePanel

[-> gamePanel: handleCombatChoice(move) (after pre-fight dialogue)
activate gamePanel

alt move == DEBUG (Pacifist)
    gamePanel -> combatChallenge **: new CombatChallenge(difficulty)
    gamePanel -> gamePanel: phaseTwoState = IN_QUIZ
    deactivate gamePanel
    
    [-> gameUI: render quiz; player picks index
    [-> gamePanel: handleQuizAnswer(index)
    activate gamePanel
    gamePanel -> combatChallenge: evaluate(index)
    combatChallenge --> gamePanel: passed (bool)
    gamePanel -> gamePanel: executeCombatStrategy(DEBUG, passed)

else move == TERMINATE (Brute Force)
    gamePanel -> gamePanel: phaseTwoState = IN_TERMINATE_MINIGAME
    gamePanel -> gameUI: initTerminateMinigame()
    deactivate gamePanel
    
    [-> gameUI: dodge mini-game runs to completion
    [-> gamePanel: handleTerminateMinigame(passed)
    activate gamePanel
    gamePanel -> gamePanel: executeCombatStrategy(TERMINATE, passed)
end

' --- Strategy execution (shared) ---
gamePanel -> combatController: initiateCombat(currentEnemy)
gamePanel -> combatController: executeCombatMove(move, passed, isBoss=false)
activate combatController
combatController -> combatStrategyFactory: createStrategy(move.name(), false)
activate combatStrategyFactory
alt move == DEBUG
    combatStrategyFactory -> debugStrategy **: new DebugStrategy(new CombatChallenge())
    combatStrategyFactory --> combatController: debugStrategy
else move == TERMINATE
    combatStrategyFactory -> terminateStrategy **: new TerminateStrategy(-10)
    combatStrategyFactory --> combatController: terminateStrategy
end
deactivate combatStrategyFactory

combatController -> combatController: setStrategy(...)
combatController -> combatController: strategy.setPassed(passed)

alt strategy == DEBUG
    combatController -> debugStrategy: execute(player, enemy)
    activate debugStrategy
    alt passed
        debugStrategy -> zEnemy: takeDamage(enemy.getHp())  ' instakill
        debugStrategy -> combatResult **: new CombatResult(true, true, +15, 0)
    else !passed
        debugStrategy -> combatResult **: new CombatResult(false, true, 0, -10)
    end
    debugStrategy --> combatController: combatResult
    deactivate debugStrategy
else strategy == TERMINATE
    combatController -> terminateStrategy: execute(player, enemy)
    activate terminateStrategy
    alt passed
        terminateStrategy -> zEnemy: takeDamage(30)
    end
    terminateStrategy -> zEnemy: isDefeated()
    alt defeated
        terminateStrategy -> combatResult **: new CombatResult(true, false, -10, 0)
    else still alive
        terminateStrategy -> combatResult **: new CombatResult(false, false, -10, passed?-10:-15)
    end
    terminateStrategy --> combatController: combatResult
    deactivate terminateStrategy
end
combatController --> gamePanel: combatResult
deactivate combatController

' --- Apply karma + outcome (in GamePanel, not CombatController) ---
gamePanel -> playerStats: updateKarma(result.getKarmaChange())
alt karma > 0
    gamePanel -> karmaTracker: add(delta, "Debug - Pacifist")
else karma < 0
    gamePanel -> karmaTracker: deduct(|delta|, "Terminate - Brute Force")
end

alt result.isVictory()
    gamePanel -> gamePanel: defeatedZombies.add(internalName)
    gamePanel -> gamePanel: checkZombieDrops(internalName)\n(adds KeyItem(1)/(2) to inventory if applicable)
    gamePanel -> gamePanel: phaseTwoState = EXPLORE
    gamePanel -> gameUI: startDialogue(victoryMsg) or post-fight cutscene
    gamePanel -> gamePanel: gameState = dialogueState
else !victory
    gamePanel -> gameUI: combatDodgePending = true (counter-attack dodge)
    gamePanel -> gamePanel: phaseTwoState = IN_COMBAT
end
destroy combatResult

' --- Player HP zero (game over) ---
opt player.takeDamage drops hp to 0 anywhere in update flow
    gamePanel -> playerEntity: isAlive()
    playerEntity --> gamePanel: false
    gamePanel -> gamePanel: gameState = gameOverState
end
deactivate gamePanel
