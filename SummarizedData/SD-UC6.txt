@startuml
title UC6: Interact with NPC (SD) - CampusFlex Section D

' â”€â”€ Reflects actual code flow â”€â”€
' NPC interactions are detected in GamePanel.checkInteractions via
' findNearbyNamedFurniture("librarian"|"haider_ramzan"|
' "cafeteria_uncle"|"server_room_guard", radius). Each handler picks a
' randomized line from a hardcoded array and calls
' UI.startDialogue / specialized cutscene methods.
'
' DialogueManager / DialogueFactory / Dialogue / DialogueOption /
' entity.NPC classes exist but are NOT used at runtime â€” the project
' does not run karma-based response selection through them.
' Easter-egg lines are gated by a per-NPC interaction counter
' (haiderInteractionCount >= 6) inside GamePanel.

participant ":GamePanel" as gamePanel
participant ":KeyHandler" as keyH
participant ":ObjectManager" as objM
participant ":Furniture" as npcF
participant ":UI" as gameUI
participant ":GameSession" as gameSession
participant ":PlayerStats" as playerStats

[-> gamePanel: checkInteractions()
activate gamePanel

alt currentZone == LIBRARY
    gamePanel -> gamePanel: interactWithLibrarian()
    gamePanel -> objM: findNearbyNamedFurniture("librarian", 3*tile)
    objM --> gamePanel: librarianFurniture
    gamePanel -> gameUI: startDialogue(random librarianDialogues[i])

else currentZone == GROUND or WALKWAY
    gamePanel -> gamePanel: interactWithHaiderRamzan()
    gamePanel -> objM: findNearbyNamedFurniture("haider_ramzan", 3*tile)
    objM --> gamePanel: haiderFurniture
    gamePanel -> gamePanel: haiderInteractionCount++
    
    alt haiderInteractionCount >= 6 AND not yet shown
        gamePanel -> gamePanel: easterEggTriggered = true
        gamePanel -> gameUI: startEasterEggDialogue("Haider Ramzan")
    else normal line
        gamePanel -> gameUI: startDialogue(random haiderDialogues[i])
    end

else currentZone == CAFETERIA AND not zombieMode
    gamePanel -> gamePanel: interactWithCafeteriaUncle()
    gamePanel -> objM: findNearbyNamedFurniture("cafeteria_uncle", 3*tile)
    objM --> gamePanel: uncleFurniture
    gamePanel -> gameUI: startCafeteriaUncleSequence()

else currentZone == SERVER_ROOM
    gamePanel -> gamePanel: interactWithServerRoomGuard()
    gamePanel -> objM: findNearbyNamedFurniture("server_room_guard", 4*tile)
    objM --> gamePanel: guardFurniture
    gamePanel -> gameUI: startServerRoomGuardSequence()
    
    opt Player tries to touch a server rack/desk
        gamePanel -> gameUI: startServerRoomGuard{Rack|Desk}Sequence()
        gamePanel -> gameSession: getPlayer().getStats()
        gamePanel -> playerStats: updateStress(+10)
    end
end

gamePanel -> gamePanel: gameState = dialogueState
deactivate gamePanel


@enduml
