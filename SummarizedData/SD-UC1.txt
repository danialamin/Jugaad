@startuml
title UC1: Start Game (SD) - CampusFlex Section D

participant ":Game" as game
participant ":GameSession" as gameSession
participant ":Player" as playerEntity
participant ":PlayerStats" as playerStats
participant ":KarmaTracker" as karmaTracker
participant ":CampusMap" as campusMap
participant ":Block" as block
participant ":Zone" as zone
participant ":GameModeFactory" as gameModeFactory
participant ":NormalMode" as normalMode
participant ":PersistenceFactory" as persistenceFactory
participant ":PersistenceHandler" as persistenceHandler

alt Player selects New Game
    [-> game: start()
    activate game
    game -> gameSession **: create()
    
    game -> gameSession: startNewGame()
    activate gameSession
    
    gameSession -> playerEntity **: create(name, spawnZoneId=1)
    activate playerEntity
    playerEntity -> playerStats **: create()
    playerStats --> playerEntity: statsRef
    playerEntity --> gameSession: playerRef
    deactivate playerEntity
    
    gameSession -> karmaTracker **: create()
    karmaTracker --> gameSession: karmaTrackerRef
    
    gameSession -> campusMap **: create()
    activate campusMap
    campusMap -> block **: createBlock(1, "C-Block")
    activate block
    block -> zone **: createZone(ZoneType.CAFETERIA, "Cafeteria")
    zone --> block: zoneRef
    block --> campusMap: blockRef
    deactivate block
    
    campusMap -> block **: createBlock(2, "D-Block")
    activate block
    block -> zone **: createZone(ZoneType.CLASSROOM, "Classroom")
    zone --> block: zoneRef
    block --> campusMap: blockRef
    deactivate block
    
    campusMap --> gameSession: campusMapRef
    deactivate campusMap
    
    gameSession -> gameModeFactory: createNormalMode()
    activate gameModeFactory
    gameModeFactory -> normalMode **: create()
    normalMode --> gameModeFactory: normalModeRef
    gameModeFactory --> gameSession: normalModeRef
    deactivate gameModeFactory
    
    gameSession -> normalMode: activate()
    gameSession -> playerEntity: setCurrentZoneId(1)
    deactivate gameSession
    deactivate game
    
else Player selects Load Game
    [-> game: loadGame(saveId)
    activate game
    game -> gameSession **: create()
    game -> gameSession: loadFromSave(saveId)
    activate gameSession
    
    gameSession -> persistenceFactory: getPersistenceHandler(type)
    activate persistenceFactory
    persistenceFactory --> gameSession: persistenceHandlerRef
    deactivate persistenceFactory
    
    gameSession -> persistenceHandler: load(saveId)
    activate persistenceHandler
    persistenceHandler --> gameSession: gameStateRef
    deactivate persistenceHandler
    
    gameSession --> game: gameStateRef
    deactivate gameSession
    deactivate game
    
    game -> gamePanel: restoreFromState(gameStateRef)
    activate gamePanel
    gamePanel -> gamePanel: currentZone = gameState.currentZoneName
    gamePanel -> gamePanel: zombieMode = gameState.zombieMode
    gamePanel -> gamePanel: phaseOneState = gameState.phaseOneState
    gamePanel -> gamePanel: phaseTwoState = gameState.phaseTwoState
    gamePanel -> gamePanel: defeatedZombies = gameState.defeatedZombies
    gamePanel -> gamePanel: player.position = (gameState.posX, gameState.posY)
    gamePanel -> gamePanel: player.stats = (hp, gpa, energy, stress, karma)
    gamePanel -> tileM: loadMap()
    gamePanel -> objM: loadZoneObjects(currentZone, zombieMode)
    gamePanel -> soundM: playZoneMusic(currentZone)
    deactivate gamePanel
end

@enduml
