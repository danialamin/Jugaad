alt Player selects "New Game" on title screen
    [-> gamePanel: keyH.enterPressed (commandNum = 0)
    activate gamePanel
    gamePanel -> gamePanel: gameState = playState
    deactivate gamePanel

else Player selects "Load Game" on title screen
    [-> gamePanel: keyH.enterPressed (commandNum = 1)
    activate gamePanel
    gamePanel -> gameSession: loadFromSave(saveId = 1)
    activate gameSession
    gameSession -> gameSession: showOptionDialog("SQL Server / XML")
    
    alt User picks SQL Server
        gameSession -> persistenceFactory: getPersistenceHandler(SQL_SERVER)
        persistenceFactory -> sqlHandler **: new SQLServerHandler()
        sqlHandler -> sqlHandler: ensureDatabaseExists()\n(JDBC create campusFlexDb if missing)
        persistenceFactory --> gameSession: sqlHandlerRef
        gameSession -> sqlHandler: load(saveId)
        sqlHandler -> hibernateUtil: getSessionFactory()
        activate hibernateUtil
        hibernateUtil --> sqlHandler: SessionFactory
        deactivate hibernateUtil
        sqlHandler -> sqlHandler: session.get(GameState.class, saveId)
        sqlHandler --> gameSession: GameState
    else User picks XML
        gameSession -> persistenceFactory: getPersistenceHandler(XML)
        persistenceFactory -> xmlHandler **: new XMLPersistenceHandler()
        gameSession -> xmlHandler: load(saveId)
        xmlHandler -> xmlHandler: XMLDecoder.readObject() from savegame.xml
        xmlHandler --> gameSession: GameState
    end

    gameSession -> playerEntity: moveTo(state.posX, state.posY)
    gameSession -> playerEntity: setHp / setMaxHp
    gameSession -> playerStats: setGPA / setEnergy / setStress / setKarma
    gameSession -> gameModeFactory: createNormalMode() OR createZombieMode()\n(based on state.modeSnapshot)
    gameSession --> gamePanel: GameState
    deactivate gameSession

    gamePanel -> gamePanel: restoreFromState(gameState)
    gamePanel -> gamePanel: currentZone = ZoneType.valueOf(state.currentZoneName)
    gamePanel -> gamePanel: zombieMode/phaseOneState/phaseTwoState/\nphase2IntroPart/defeatedZombies/\nserverBossHp/prayedThisPhase â† state
    gamePanel -> tileM: loadMap()
    gamePanel -> objM: clearObjects() + load<Zone>Objects()\n(zombie variant if zombieMode)
    gamePanel -> objM: filterDefeatedZombies(defeatedZombies)
    gamePanel -> soundManager: setZombieMode(zombieMode)
    gamePanel -> soundManager: playZoneMusic(currentZone)
    gamePanel -> gamePanel: gameState = playState
    deactivate gamePanel
end
