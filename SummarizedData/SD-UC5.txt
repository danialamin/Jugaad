participant ":GamePanel" as gamePanel

[-> gamePanel: checkInteractions()\n
activate gamePanel

gamePanel -> gamePanel: interactWithOptionalPrayer()
gamePanel -> objM: findNearbyNamedFurniture("empty_prayer_mat", 2*tile)

alt mat found
    objM --> gamePanel: matRef
    
    alt prayedThisPhase == false (Prayer Area available)
        gamePanel -> gamePanel: prayedThisPhase = true
        gamePanel -> gameSession: getPlayer()
        gameSession --> gamePanel: playerRef
        gamePanel -> playerEntity: getStats()
        gamePanel -> playerStats: updateKarma(+20)
        activate playerStats
        playerStats -> playerStats: karma += 20
        deactivate playerStats
        
        gamePanel -> gameSession: getKarmaTracker()
        gameSession --> gamePanel: karmaTrackerRef
        gamePanel -> karmaTracker: add(20, "Prayer at Prayer Area")
        activate karmaTracker
        karmaTracker -> karmaTracker: history.add(KarmaEvent(+20, ...))\ntotal += 20
        deactivate karmaTracker
        
        gamePanel -> gameUI: startDialogue("You|You pray quietly...")
        
    else prayedThisPhase == true (already prayed)
        gamePanel -> gameUI: startDialogue("You|You already prayed...")
    end
    
    gamePanel -> gamePanel: gameState = dialogueState
