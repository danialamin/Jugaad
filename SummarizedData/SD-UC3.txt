title UC3: Use Phone (SD) - CampusFlex Section D

' â”€â”€ Reflects actual code flow â”€â”€
' Phone is opened from GamePanel.update() via keyH.pPressed.
' All phone-menu logic lives in UI.updatePhoneMenu, which decides
' between cheat (during quiz) and social-media / close / open-Flex
' based on session.isLectureQuizActive() and the selected option.

participant ":GamePanel" as gamePanel
participant ":KeyHandler" as keyH
participant ":GameSession" as gameSession
participant ":Player" as playerEntity
participant ":Phone" as phone
participant ":UI" as gameUI
participant ":PlayerStats" as playerStats
participant ":KarmaTracker" as karmaTracker
participant ":StatModifierImpl" as statModifierImpl

[-> gamePanel: update()
activate gamePanel
gamePanel -> playerEntity: getPhone()
playerEntity --> gamePanel: phoneRef
gamePanel -> phone: open()
gamePanel -> gameUI: openPhoneMenu()
gamePanel -> gamePanel: gameState = phoneState
deactivate gamePanel

[-> gameUI: updatePhoneMenu(session)
activate gameUI
gameUI -> phone: isDuringQuiz(session)
phone -> gameSession: isLectureQuizActive()
gameSession --> phone: bool
phone --> gameUI: bool

alt Phone opened DURING quiz (cheat path)
    gameUI -> phone: buildCheatModifier()
    phone -> statModifierImpl **: new StatModifierImpl(0, 0, -15)
    phone --> gameUI: cheatModifier
    
    gameUI -> playerStats: applyModifier(cheatModifier)
    activate playerStats
    playerStats -> playerStats: updateGPA(0)\nupdateStress(0)\nupdateKarma(-15)
    deactivate playerStats
    
    gameUI -> karmaTracker: deduct(15, "Phone during quiz - cheating")
    gameUI -> phone: close()
    gameUI -> gameUI: quizAiAssistPath = true
    gameUI -> gameUI: startDialogue(teacher catches cheating)
    gameUI -> gamePanel: gameState = dialogueState
    destroy statModifierImpl

else Phone opened OUTSIDE quiz
    gameUI -> keyH: read up/down/enter
    
    alt Option 0 â€” Scroll socials
        gameUI -> phone: buildSocialMediaModifier()
        phone -> statModifierImpl **: new StatModifierImpl(0, -5, -5)
        phone --> gameUI: socialModifier
        
        gameUI -> playerStats: applyModifier(socialModifier)
        activate playerStats
        playerStats -> playerStats: updateGPA(0)\nupdateStress(-5)\nupdateKarma(-5)
        deactivate playerStats
        
        gameUI -> karmaTracker: deduct(5, "Scrolled social media")
        gameUI -> phone: close()
        gameUI -> gamePanel: gameState = playState
        destroy statModifierImpl
        
    else Option 1 â€” Close phone
        gameUI -> phone: close()
        gameUI -> gamePanel: gameState = playState
        
    else Option 2 â€” Open Flex (no stat change)
        gameUI -> gameUI: startDialogue("You|Flex opens after thinking about it...")
        gameUI -> phone: close()
        gameUI -> gamePanel: gameState = dialogueState
        
    else Option 3 â€” Stress display only
    end
end
deactivate gameUI

@enduml
