' --- Door transition detection ---
gamePanel -> campusMap: getZone(currentZone)
campusMap --> gamePanel: zoneRef
gamePanel -> zone: getLocations()
zone --> gamePanel: List of Location

loop each Location in current Zone
    gamePanel -> location: getPosition / getRequiredDirection
    alt player tile == door tile AND facing requiredDirection
        location --> gamePanel: targetZone, targetSpawnPosition
        
        alt (zombieMode and locked-door rules)
            alt target == CORRIDOR AND !inventory.hasItem(1)
                gamePanel -> gameUI: startDialogue("ACCESS DENIED. Corridor Keycard.")
                gamePanel -> gamePanel: pushPlayerBack(dir); gameState = dialogueState
            else target == SERVER_ROOM AND !inventory.hasItem(2)
                gamePanel -> gameUI: startDialogue("ACCESS DENIED. Server Access Card.")
                gamePanel -> gamePanel: pushPlayerBack(dir); gameState = dialogueState
            else target == AI_LAB (zombie mode)
                gamePanel -> gameUI: startDialogue("AI Lab is locked tight.")
                gamePanel -> gamePanel: pushPlayerBack(dir); gameState = dialogueState
            end
        end
        
        gamePanel -> gamePanel: currentZone = targetZone
        gamePanel -> tileM: loadMap()
        gamePanel -> objM: clearObjects() + load[Zone]Objects()\n(zombie variant if zombieMode)
        gamePanel -> objM: filterDefeatedZombies(defeatedZombies)
        gamePanel -> playerEntity: setXLocation/YLocation(targetSpawnPosition)
        gamePanel -> playerEntity: setCurrentZoneId(currentZone.ordinal())
        gamePanel -> soundManager: playZoneMusic(currentZone)
        
        opt Special arrival cutscene\n(CLASSROOM/AI_LAB lecture intro,\nzombie cafe/classroom intro,\nServer Room arrival)
            gamePanel -> gameUI: start...Sequence()
            gamePanel -> gamePanel: gameState = dialogueState
        end
    end
end
