@startuml
title UC8: Explore Campus Map (SD) - CampusFlex Section D

participant ":GameSession" as gameSession
participant ":GameMode" as gameMode
participant ":Player" as playerEntity
participant ":CampusMap" as campusMap
participant ":Zone" as zone
participant ":SoundManager" as soundManager

[-> gameSession: handleSystemEvent("exploreCampus")
activate gameSession
gameSession -> gameMode: onPlayerMove(x, y, session)
activate gameMode

gameMode -> playerEntity: moveTo(x, y)
activate playerEntity
playerEntity --> gameMode: positionUpdated
deactivate playerEntity

gameMode -> campusMap: getZone(player.getCurrentZoneId())
activate campusMap
campusMap --> gameMode: currentZoneRef
deactivate campusMap

alt Player reaches door Location
    gameMode -> zone: getNearbyLocation(player.getPosition())
    activate zone
    zone --> gameMode: doorLocationRef
    deactivate zone
    
    gameMode -> zone: getLocations()
    note right: Location holds targetZone : ZoneType
    
    gameMode -> gameSession: updateZone(nextZone)
    
    gameSession -> soundManager: playZoneMusic(nextZone.getType())
    activate soundManager
    soundManager -> soundManager: stopMusic()
    soundManager -> soundManager: playZoneMusic(zoneType)
    soundManager --> gameSession: musicPlaying
    deactivate soundManager
    
    gameMode --> gameSession: transitionComplete
end

gameSession --> [: renderUpdatedMapAndMusic()
deactivate gameMode
deactivate gameSession
