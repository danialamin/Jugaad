@startuml
title UC7: Save Game (SD) - CampusFlex Section D

participant ":GamePanel" as gamePanel
participant ":GameSession" as gameSession
participant ":SavePoint" as savePoint
participant ":PersistenceFactory" as persistenceFactory
participant ":PersistenceHandler" as persistenceHandler
participant ":GameState" as gameState

[-> gamePanel: checkInteractions()
activate gamePanel
gamePanel -> savePoint: detect nearby checkpoint
activate savePoint
savePoint --> gamePanel: checkpoint found
deactivate savePoint

gamePanel -> gamePanel: buildFullGameState()
gamePanel -> gameState **: create(saveId, posX, posY, currentZoneName,\nmodeSnapshot, zombieMode, phaseOneState, phaseTwoState,\ndefeatedZombies, serverBossHp, hp, gpa, energy, stress, karma)

gamePanel -> gameSession: saveGameState(gameState)
activate gameSession

alt Player cancels save
    [-> gameSession: handleSystemEvent("cancelSave")
else Player confirms save
    [-> gameSession: handleSystemEvent("confirmSave")
    
    gameSession -> persistenceFactory: getPersistenceHandler(type)
    activate persistenceFactory
    persistenceFactory --> gameSession: persistenceHandlerRef
    deactivate persistenceFactory
    
    gameSession -> persistenceHandler: save(gameState)
    activate persistenceHandler
    
    alt Database write SUCCEEDS
        persistenceHandler --> gameSession: isSaved = true
    else Database write FAILS
        persistenceHandler --> gameSession: isSaved = false
    end
    deactivate persistenceHandler
end
deactivate gameSession
destroy gameState
deactivate gamePanel
@enduml
