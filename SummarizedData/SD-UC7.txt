' Save is triggered when GamePanel.checkInteractions finds a
' "checkpoint" furniture near the player. GamePanel builds the full
' GameState snapshot via buildFullGameState() and asks GameSession
' to persist it. GameSession shows a JOptionPane to pick SQL Server
' (Hibernate-backed SQLServerHandler) or File System (XMLPersistenceHandler).
' Save IDs are hardcoded to 1 (single save slot).

participant ":GamePanel" as gamePanel
participant ":KeyHandler" as keyH
participant ":ObjectManager" as objM
participant ":Furniture" as cpF
participant ":GameState" as gameState
participant ":GameSession" as gameSession
participant ":Player" as playerEntity
participant ":PlayerStats" as playerStats
participant ":KarmaTracker" as karmaTracker
participant ":PersistenceFactory" as persistenceFactory
participant ":SQLServerHandler" as sqlHandler
participant ":HibernateUtil" as hibernateUtil
participant "session:Session" as hbSession
participant ":XMLPersistenceHandler" as xmlHandler
participant ":UI" as gameUI

[-> gamePanel: checkInteractions()
activate gamePanel
gamePanel -> objM: findNearbyNamedFurniture("checkpoint", 2*tile)
objM --> gamePanel: cpF
gamePanel -> gamePanel: checkpointSave()

gamePanel -> gamePanel: buildFullGameState()
gamePanel -> gameState **: new GameState()
activate gameState
gamePanel -> gameState: setSaveId(1)
gamePanel -> gameState: setPosX/Y, setCurrentZoneName,\nsetModeSnapshot, setZombieMode,\nsetPhaseOneState/PhaseTwoState/Phase2IntroPart,\nsetDefeatedZombies, setServerBossHp, setPrayedThisPhase
gamePanel -> playerEntity: getHp / getMaxHp
gamePanel -> playerStats: getGPA / getEnergy / getStress / getKarma
gamePanel -> gameState: setHp/setMaxHp/setGpa/\nsetEnergy/setStress/setKarma/setTimestamp

gamePanel -> gameSession: saveGameState(gameState)
activate gameSession
gameSession -> gameSession: showOptionDialog("SQL Server" | "File System (XML)")

alt Player closes dialog (cancel)
    gameSession -> gameSession: handleSystemEvent("cancelSave")
    gameSession --> gamePanel: (no write)

else Player confirms
    gameSession -> gameSession: handleSystemEvent("confirmSave")
    
    alt Choice == 0 (SQL Server / Hibernate)
        gameSession -> persistenceFactory: getPersistenceHandler(SQL_SERVER)
        persistenceFactory -> sqlHandler **: new SQLServerHandler()
        sqlHandler -> sqlHandler: ensureDatabaseExists()\n(JDBC create campusFlexDb)
        persistenceFactory --> gameSession: handlerRef
        gameSession -> sqlHandler: save(gameState)
        sqlHandler -> hibernateUtil: getSessionFactory()
        activate hibernateUtil
        hibernateUtil --> sqlHandler: SessionFactory
        deactivate hibernateUtil
        sqlHandler -> hbSession **: openSession()
        sqlHandler -> hbSession: beginTransaction()
        sqlHandler -> hbSession: merge(gameState)
        sqlHandler -> hbSession: commit()
        sqlHandler -> hbSession: close()
        destroy hbSession
        sqlHandler --> gameSession: success boolean
        
    else Choice == 1 (File System / XML)
        gameSession -> persistenceFactory: getPersistenceHandler(XML)
        persistenceFactory -> xmlHandler **: new XMLPersistenceHandler()
        gameSession -> xmlHandler: save(gameState)
        xmlHandler -> xmlHandler: XMLEncoder.writeObject(state)\nâ†’ savegame.xml
        xmlHandler --> gameSession: success boolean
    end
    
    gameSession -> gameSession: showMessageDialog("Saved!" / "Failed.")
end
deactivate gameSession

gamePanel -> gameUI: startDialogue("System|Checkpoint reached. Progress saved.")
gamePanel -> gamePanel: gameState = dialogueState
destroy gameState
deactivate gamePanel
