participant ":GamePanel" as gamePanel
participant ":KeyHandler" as keyH
participant ":ObjectManager" as objM
participant ":Furniture" as desk
participant ":Player" as playerEntity
participant ":UI" as gameUI
participant ":GameSession" as gameSession
participant ":Phone" as phone
participant ":PlayerStats" as playerStats
participant ":KarmaTracker" as karmaTracker
participant ":StatModifierImpl" as statModifierImpl

[-> gamePanel: update()
activate gamePanel
gamePanel -> gamePanel: checkInteractions()
gamePanel -> keyH: read ePressed / enterPressed

alt Pre-condition: in CLASSROOM (CS_CLASS_REQUIRED)\n         OR in AI_LAB (AI_LAB_REQUIRED)
    gamePanel -> objM: scan furnitureList near player
    gamePanel -> desk: name == "empty_desk" within 1.5 tiles
    activate desk
    desk --> gamePanel: deskRef
    deactivate desk
    
    gamePanel -> playerEntity: setXLocation/YLocation/direction (sit on desk)
    gamePanel -> playerEntity: setSeatedInClass(true)
    gamePanel -> gamePanel: beginLectureForCurrentZone()\n(activeLectureZone = currentZone)
    gamePanel -> gameUI: startClassroomLectureSequence()\nOR startAILectureSequence()
    gamePanel -> gamePanel: gameState = dialogueState
    deactivate gamePanel
    
    [-> gameUI: updateDialogueScreen() â€” pages through lecture
    activate gameUI
    gameUI -> gameUI: when lecture queue ends, transitionToQuiz = true
    gameUI -> gameSession: setLectureQuizActive(true)
    gameUI -> gameUI: beginQuizEncounter()
    gameUI -> gamePanel: gameState = quizState
    deactivate gameUI
    
    alt Player opens phone DURING quiz (cheat)
        [-> gamePanel: keyH.pPressed && session.isLectureQuizActive()
        activate gamePanel
        gamePanel -> playerEntity: getPhone()
        gamePanel -> phone: open()
        gamePanel -> gameUI: openPhoneMenu()
        gamePanel -> gamePanel: gameState = phoneState
        deactivate gamePanel
        
        [-> gameUI: updatePhoneMenu(session)
        activate gameUI
        gameUI -> phone: isDuringQuiz(session)
        phone --> gameUI: true
        gameUI -> phone: buildCheatModifier()
        phone -> statModifierImpl **: new StatModifierImpl(0, 0, -15)
        phone --> gameUI: cheatModifier
        gameUI -> playerStats: applyModifier(cheatModifier)
        playerStats -> playerStats: updateGPA(0)\nupdateStress(0)\nupdateKarma(-15)
        gameUI -> karmaTracker: deduct(15, "Phone during quiz - cheating")
        gameUI -> phone: close()
        gameUI -> gameUI: quizAiAssistPath = true\n(post-quiz GPA/Karma penalty marked)
        gameUI -> gameUI: startDialogue(teacher reaction)
        deactivate gameUI
        destroy statModifierImpl
        
    else Player answers honestly (submitAnswer)
        [-> gameUI: keyH.enterPressed in quizState
        activate gameUI
        gameUI -> gameUI: evaluate(quizCommandNum == correctIndex)
        
        alt Answer CORRECT
            gameUI -> playerStats: updateGPA(+0.3)
            gameUI -> playerStats: updateKarma(+5)
            gameUI -> karmaTracker: add(5, "Correct quiz answer")
        else Answer WRONG
            gameUI -> playerStats: updateGPA(+0.1)
        end
        
        gameUI -> gameSession: setLectureQuizActive(false)
        gameUI -> gamePanel: completeCurrentLectureQuiz()
        gameUI -> gamePanel: gameState = playState
        deactivate gameUI
    end

else Player skips class entirely
end
